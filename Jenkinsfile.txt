pipeline {
    agent any

    stages {
        stage('Clone repository') {
            steps {
                sh ('git clone https://github.com/divya09jan/Learning.git')
				sh ('mvn clean -f Learning.git')
            }
        }
        stage('Terraform Init') {
            steps {
                sh ('terraform init')
            }
        }
        stage('Terraform Apply') {
            steps {
                sh ('terraform apply -auto-approve')
            }
        }
    }
}
